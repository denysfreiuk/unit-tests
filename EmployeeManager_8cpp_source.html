126 auto aviaryIt = zooGraph.getAviaries().find(aviaryId);

131
132 auto empIt = employees.find(employeeId);

134 logger.error("Employee with ID " + employeeId + " not found.");

137
138 auto aviaryPtr = std::dynamic_pointer_cast<Aviary>(aviaryIt->second);

143
144 std::string assignedId = aviaryPtr->getAssignedEmployee();
145 if (assignedId.empty()) {

147 return false;

152 return false;

156 empIt->second->removeAviary(aviaryId);
157
158 employeeRepo.removeEmployeeFromAviary(empIt->second->getId(), aviaryId);

164bool EmployeeManager::removeEmployee(const string& id) {
165 logger.debug("Removing employee with ID " + id + ".");
166
167 auto emp = getEmployee(id);
168 if (!emp) {
169 logger.warn("No employee with ID " + id + " found.");
170 return false;
171 }
172
173
174 for (auto& [aviaryId, aviaryPtr] : zooGraph.getAviaries()) {
175 auto realAviaryPtr = std::dynamic_pointer_cast<Aviary>(aviaryPtr);
176 if (!realAviaryPtr) {
177 logger.warn("Vertex " + aviaryId + " is not an aviary.");
178 continue;
179 }
180
181 if (realAviaryPtr->getAssignedEmployee() == id) {
182 realAviaryPtr->removeAssignedEmployee();
183 logger.info("Employee " + emp->getName() + " was removed from aviary " + realAviaryPtr->getName() + ".");
184 }
185 }
186
187 employees.erase(id);
188 employeeRepo.removeEmployee(id);
189 logger.info("Employee " + emp->getName() + " removed from system.");
190 return true;
191}
