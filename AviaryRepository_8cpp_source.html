79 if (sqlite3_prepare_v2(db.getDB(), sql.c_str(), -1, &stmt, nullptr) != SQLITE_OK) {
80 cerr << "SQL prepare error: " << sqlite3_errmsg(db.getDB()) << endl;
81 return aviaries;
82 }
83
84 auto getTextSafe = [&](int col) -> string {
85 const unsigned char* txt = sqlite3_column_text(stmt, col);
86 return txt ? reinterpret_cast<const char*>(txt) : "";
87 };
88
89 while (sqlite3_step(stmt) == SQLITE_ROW) {

94 int capacity = sqlite3_column_int(stmt, 4);
95 string animals = getTextSafe(5);

99 aviaries[id] = aviary;

103 }

105 sqlite3_finalize(stmt);
