91void Employee::setAssignedAviaries(const string& assignedAviaries) {
92 logger.debug("setAssignedAviaries input: [" + assignedAviaries + "]");
93 aviaryIds.clear();
94
95 if (assignedAviaries.empty()) return;
96
97 stringstream ss(assignedAviaries);
98 string aviaryId;
99
100 while (getline(ss, aviaryId, ',')) {
101 size_t start = aviaryId.find_first_not_of(" \t");
102 size_t end = aviaryId.find_last_not_of(" \t");
103 if (start != string::npos && end != string::npos)
104 aviaryId = aviaryId.substr(start, end - start + 1);

106 aviaryId.clear();
107
108 if (!aviaryId.empty())
109 aviaryIds.push_back(aviaryId);
110 }
111

162void Employee::replaceAviary(const string& fromAviary, const string& toAviary) {
