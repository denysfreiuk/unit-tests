20Animal::Animal(const string& name, const string& species, int age, double weight, const string& type)
21 : name(name), species(species), age(age), weight(weight), type(type) {
22
23 UUID uuid;
24 RPC_STATUS status = UuidCreate(&uuid);
25 if (status != RPC_S_OK && status != RPC_S_UUID_LOCAL_ONLY && status != RPC_S_UUID_NO_ADDRESS) {
26 logger.error("UuidCreate failed, RPC_STATUS = " + to_string(status));
27 throw runtime_error("UuidCreate failed, RPC_STATUS = " + to_string(status));
28 }
29
30 RPC_CSTR strUuid = nullptr;
31 status = UuidToStringA(&uuid, &strUuid);
32 if (status != RPC_S_OK) {
33 logger.error("UuidToString failed, RPC_STATUS = " + to_string(status));
34 throw runtime_error("UuidToString failed, RPC_STATUS = " + to_string(status));
35 }
36
37 id = string(reinterpret_cast<char*>(strUuid));
38 RpcStringFreeA(&strUuid);
39
40 logger.info("Created Animal: " + name + " (" + species + "), ID: " + id);
41}

122bool Animal::isCompatibleWith(const shared_ptr<Animal>& other) const {
123 logger.debug("Checking compatibility between " + name + " (" + species + ") and " + other->name + " (" + other->species + ")");
124
125 if ((species == "Lion" && other->species == "Tiger") ||
126 (species == "Tiger" && other->species == "Lion") ||
127 (species == "Wolf" && other->species == "Bear") ||
128 (species == "Bear" && other->species == "Wolf")) {
129 logger.warn("Incompatible: predator species conflict (" + species + " vs " + other->species + ")");
130 return false;
131 }
132
133 if ((species == "Eagle" && other->species == "Parrot") ||
134 (species == "Parrot" && other->species == "Eagle") ||
135 (species == "Owl" && other->species == "Crow") ||
136 (species == "Crow" && other->species == "Owl")) {
137 logger.warn("Incompatible: bird conflict (" + species + " vs " + other->species + ")");
138 return false;
139 }
140
141 if ((species == "Snake" && (other->type == "Mammal" || other->type == "Bird")) ||
142 ((other->species == "Snake") && (type == "Mammal" || type == "Bird"))) {
143 logger.warn("Incompatible: reptile vs mammal/bird (" + species + " vs " + other->species + ")");
144 return false;
145 }
146
147 if ((type == "Fish" && other->type == "Fish") &&
148 ((species == "Piranha") || (other->species == "Piranha"))) {
149 logger.warn("Incompatible: aggressive fish detected (" + species + " vs " + other->species + ")");
150 return false;
151 }
152
153 if ((type == "Amphibian" && other->type == "Insect") ||
154 (type == "Insect" && other->type == "Amphibian")) {
155 logger.warn("Incompatible: amphibian vs insect (" + species + " vs " + other->species + ")");
156 return false;
157 }
158
159 if ((type == "Arachnid") &&
160 (other->type == "Insect" || other->type == "Amphibian" || other->type == "Fish")) {
161 logger.warn("Incompatible: arachnid vs small species (" + species + " vs " + other->species + ")");
162 return false;
163 }
164
165 logger.info("Animals " + name + " and " + other->name + " are compatible");
