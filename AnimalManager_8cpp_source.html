52void AnimalManager::linkAnimalsToAviaries() {
53 auto& aviaries = zooGraph.getAviaries();
54
55 for (const auto& [id, vertex] : aviaries) {
56 auto aviary = dynamic_pointer_cast<Aviary>(vertex);
57 if (!aviary) continue;
58
59 const string& str = aviary->getAnimalsStrTemp();
60 if (str.empty()) continue;
61
62 stringstream ss(str);
63 string animalId;
64
65 while (getline(ss, animalId, ',')) {
66 size_t start = animalId.find_first_not_of(" \t");
67 size_t end = animalId.find_last_not_of(" \t");
68 if (start == string::npos || end == string::npos) continue;
69 animalId = animalId.substr(start, end - start + 1);
70
71 auto it = animals.find(animalId);
72 if (it != animals.end()) {
73 aviary->getAnimalsRef().push_back(it->second);
74 logger.debug("Linked animal " + animalId + " to aviary " + id);
75 }
76 }
77
78 aviary->clearAnimalsStrTemp(); // більше не потрібно
79 }
80}

158bool AnimalManager::removeAnimalFromAnimals(const string& animalId) {
159 logger.debug("removeAnimalFromAnimals() called with animalId=" + animalId);
160
161 auto it = animals.find(animalId);
162 if (it == animals.end()) {
163 logger.warn("Animal not found: " + animalId);
164 return false;
165 }
166
167 auto& aviaries = zooGraph.getAviaries();
168 for (auto& [aviaryId, vertexPtr] : aviaries) {
169 auto aviaryPtr = dynamic_pointer_cast<Aviary>(vertexPtr);
170 if (!aviaryPtr) continue;
171 if (aviaryPtr->hasAnimal(animalId)) {
172 removeAnimalFromAviary(aviaryId, animalId);
173 break;
174 }
175 }
176
177 animals.erase(it);
178 animalRepo.removeAnimal(animalId);
179 logger.info("Animal " + animalId + " deleted from system.");
180 return true;
181}

183bool AnimalManager::moveAnimalBetweenAviaries(const string& fromAviaryId, const string& toAviaryId, const string& animalId) {
184 logger.debug("moveAnimalBetweenAviaries() called: from=" + fromAviaryId + ", to=" + toAviaryId + ", animalId=" + animalId);
185
186 auto fromIt = zooGraph.getAviaries().find(fromAviaryId);
187 auto toIt = zooGraph.getAviaries().find(toAviaryId);
188 if (fromIt == zooGraph.getAviaries().end() || toIt == zooGraph.getAviaries().end()) {
189 logger.error("One of the aviaries does not exist. From=" + fromAviaryId + ", To=" + toAviaryId);
190 return false;
191 }
192
193 auto fromAviary = dynamic_pointer_cast<Aviary>(fromIt->second);
194 auto toAviary = dynamic_pointer_cast<Aviary>(toIt->second);
195 if (!fromAviary || !toAviary) {
196 logger.error("Invalid aviary pointers detected.");
197 return false;
198 }
199
200 auto animalIt = animals.find(animalId);
201 if (animalIt == animals.end()) {
202 logger.error("Animal not found: " + animalId);
203 return false;
204 }
205 auto animal = animalIt->second;
206
207 if (!fromAviary->hasAnimal(animalId)) {
208 logger.warn("Animal " + animal->getName() + " not in aviary " + fromAviary->getName());
209 return false;
210 }
211
212 if (!toAviary->canAddAnimal(animal)) {
213 logger.warn("Cannot move animal \"" + animal->getName() + "\" to aviary \"" + toAviary->getName() + "\" (incompatible or full)");
214 return false;
215 }
216
217 fromAviary->removeAnimal(animalId);
218 toAviary->addAnimal(animal);
219 animalRepo.moveAnimal(animalId, fromAviaryId, toAviaryId);
220
221 logger.info("Animal \"" + animal->getName() + "\" moved from \"" + fromAviary->getName() + "\" to \"" + toAviary->getName() + "\".");
222
223 return true;
224}
