136

141
142 while (sqlite3_step(stmt) == SQLITE_ROW) {

144 string name = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 1));

146 string type = reinterpret_cast<const char*>(sqlite3_column_text(stmt, 3));

150 if (aviaryText) aviaryId = reinterpret_cast<const char*>(aviaryText);
151

153 double weight = sqlite3_column_double(stmt, 6);
154
155 shared_ptr<Animal> animal;
156
157 if (type == "Mammal")
158 animal = make_shared<Mammal>(id, name, species, age, weight, "Mammal", aviaryId);
159 else if (type == "Reptile")
160 animal = make_shared<Reptile>(id, name, species, age, weight, "Reptile", aviaryId);
161 else if (type == "Bird")
162 animal = make_shared<Bird>(id, name, species, age, weight, "Bird", aviaryId);
163 else if (type == "Fish")
164 animal = make_shared<Fish>(id, name, species, age, weight, "Fish", aviaryId);

167 else if (type == "Arachnid")

169 else if (type == "Insect")
