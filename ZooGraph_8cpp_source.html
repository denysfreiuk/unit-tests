148bool Aviary::canAddAnimal(const shared_ptr<Animal>& animal) const {
149 if (!animal) {
150 logger.warn("canAddAnimal: Tried to add a null animal pointer.");
151 return false;
152 }
153
154 logger.debug("canAddAnimal: Checking if animal [" + animal->getId() + "] can be added to aviary [" + getName() + "].");
155
156 if (hasAnimal(animal->getId())) {
157 logger.debug("canAddAnimal: Animal [" + animal->getId() + "] is already in this aviary.");
158 return false;
159 }
160
161 if (static_cast<int>(animals.size()) >= capacity) {
162 logger.debug("canAddAnimal: Aviary is full (" + std::to_string(animals.size()) + "/" + std::to_string(capacity) + ").");
163 return false;
164 }
165
166 for (const auto& existing : animals) {
167 if (!existing) {
168 logger.warn("canAddAnimal: Found a null existing animal pointer in aviary. Skipping compatibility check...");
169 continue;
170 }
171
172 if (!existing->isCompatibleWith(animal)) {
173 logger.debug("canAddAnimal: Animal [" + animal->getId() + "] is NOT compatible with existing animal [" + existing->getId() + "] (existing -> new).");
174 return false;
175 }
176
177 if (!animal->isCompatibleWith(existing)) {
178 logger.debug("canAddAnimal: Animal [" + animal->getId() + "] is NOT compatible with existing animal [" + existing->getId() + "] (new -> existing).");
179 return false;
180 }
181
182 logger.debug("canAddAnimal: Animal [" + animal->getId() + "] is compatible with existing animal [" + existing->getId() + "].");
183 }
184
185 logger.info("canAddAnimal: Animal [" + animal->getId() + "] CAN be added to aviary [" + getName() + "].");
186 return true;
187}

215void ZooGraph::loadAviariesFromRepo(AviaryRepository& repo) {

237void ZooGraph::loadPathsFromRepo(PathRepository& repo) {
238 try {
239 vector<shared_ptr<Path>> allPaths = repo.getAllPaths();
240
241 logger.info("Loaded " + to_string(allPaths.size()) + " paths from database.");
242
243 for (const auto& path : allPaths) {
244 if (!path) continue;
245
246 string fromId = path->getFromId();
247 string toId = path->getToId();
248 double length = path->getLength();
249
250 if (getVertex(fromId) && getVertex(toId)) {
251 addEdge(fromId, toId, length);
252 logger.debug("Added path: " + fromId + " -> " + toId +
253 " (length = " + to_string(length) + ")");
254 } else {
255 logger.warn("Path missed: no vertex found for " +
256 fromId + " or " + toId);
257 }
258 }
259
260 logger.info("All paths successfully added to ZooGraph.");
261 }
262 catch (const exception& e) {
263 logger.error(string("Error loading paths: ") + e.what());
264 }
