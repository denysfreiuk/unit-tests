171vector<string> Graph::findPath(const string& startId, const string& endId) {
172 logger.debug("Finding path (unweighted) from " + startId + " to " + endId);
173
174 unordered_map<string, string> parent;
175 queue<string> q;
176 unordered_map<string, bool> visited;
177
178 q.push(startId);
179 visited[startId] = true;
180
181 while (!q.empty()) {
182 string cur = q.front(); q.pop();
183 if (cur == endId) break;
184
185 for (const auto& e : edges) {
186 string neighbor;
187 if (e.getFrom() == cur) neighbor = e.getTo();
188 else if (e.getTo() == cur) neighbor = e.getFrom();
189 else continue;
190
191 if (!visited[neighbor]) {
192 visited[neighbor] = true;
193 parent[neighbor] = cur;
194 q.push(neighbor);
195 }
196 }
197 }
198
199 vector<string> path;
200 if (!visited[endId]) {
201 logger.warn("Path not found from " + startId + " to " + endId);
202 return path;
203 }
204
205 for (string v = endId; v != startId; v = parent[v]) path.push_back(v);
206 path.push_back(startId);
207 reverse(path.begin(), path.end());
208
209 logger.debug("Path found: length = " + to_string(path.size()));
210 return path;
211}

213vector<string> Graph::findPathByWeight(const string& startId, const string& endId) const {
214 logger.debug("Finding shortest path (by weight) from " + startId + " to " + endId);
215
216 unordered_map<string, double> dist;
217 unordered_map<string, string> parent;
218
219 for (auto& [id, v] : vertices) {
220 dist[id] = numeric_limits<double>::infinity();
221 }
222 dist[startId] = 0;
223
224 using P = pair<double, string>;
225 priority_queue<P, vector<P>, greater<P>> pq;
226 pq.push({0, startId});
227
228 while (!pq.empty()) {
229 auto [d, u] = pq.top(); pq.pop();
230 if (d > dist[u]) continue;
231
232 for (const auto& e : edges) {
233 string neighbor;
234 double weight = e.getWeight();
235 if (e.getFrom() == u) neighbor = e.getTo();
236 else if (e.getTo() == u) neighbor = e.getFrom();
237 else continue;
238
239 double newDist = d + weight;
240 if (newDist < dist[neighbor]) {
241 dist[neighbor] = newDist;
242 parent[neighbor] = u;
243 pq.push({newDist, neighbor});
244 }
245 }
246 }
247
248 vector<string> path;
249 if (dist[endId] == numeric_limits<double>::infinity()) {
250 logger.warn("No weighted path found between " + startId + " and " + endId);
251 return path;
252 }
253
254 for (string v = endId; v != startId; v = parent[v]) path.push_back(v);
255 path.push_back(startId);
256 reverse(path.begin(), path.end());
257
258 logger.info("Shortest weighted path found: total weight = " + to_string(dist[endId]));
259 return path;
260}

262double Graph::distanceBetween(const string& fromId, const string& toId) const {
263 logger.debug("Calculating distanceBetween() " + fromId + " <-> " + toId);
264
265 vector<string> path = findPathByWeight(fromId, toId);
266 if (path.empty()) {
267 logger.warn("No path between " + fromId + " and " + toId);
268 return -1.0;
269 }
270
271 double totalDist = 0.0;
272 for (size_t i = 0; i + 1 < path.size(); ++i) {
273 for (const auto& e : edges) {
274 if ((e.getFrom() == path[i] && e.getTo() == path[i + 1]) ||
275 (e.getTo() == path[i] && e.getFrom() == path[i + 1])) {
276 totalDist += e.getWeight();
277 break;
278 }
279 }
280 }
281
282 logger.info("Total distance between " + fromId + " and " + toId + " = " + to_string(totalDist));
283 return totalDist;
284}

286bool Graph::checkConnectivity() const {
287 logger.debug("Checking connectivity of graph");
288
289 if (vertices.empty()) {
290 logger.warn("Graph is empty, considered connected by default");
291 return true;
292 }
293
294 unordered_map<string, bool> visited;
295 queue<string> q;
296
297 string startId = vertices.begin()->first;
298 q.push(startId);
299 visited[startId] = true;
300
301 while (!q.empty()) {
302 string cur = q.front(); q.pop();
303 for (const auto& e : edges) {
304 string neighbor;
305 if (e.getFrom() == cur) neighbor = e.getTo();
306 else if (e.getTo() == cur) neighbor = e.getFrom();
307 else continue;
308
309 if (!visited[neighbor]) {
310 visited[neighbor] = true;
311 q.push(neighbor);
312 }
313 }
314 }
315
316 bool connected = (visited.size() == vertices.size());
317 logger.debug(string("Graph connectivity check result: ") + (connected ? "Connected" : "Disconnected"));
